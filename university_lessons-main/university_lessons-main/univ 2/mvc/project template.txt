mkdir backend
cd backend
npm init -y
npm install express

##Create a basic Express server: Create a file named server.js:

const express = require('express');
const app = express();
const PORT = 5000;

app.get('/', (req, res) => {
    res.send('Hello from the backend!');
});

app.listen(PORT, () => {
    console.log(`Server is running on http://localhost:${PORT}`);
});


##Start the backend server: Add a start script in package.json:

"scripts": {
    "start": "node server.js"
}

Now, start the backend server:

npm start
